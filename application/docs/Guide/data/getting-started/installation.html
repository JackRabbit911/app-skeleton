{{ __('Installation') }}

{{ __('The easiest way') }}:

    git clone https://github.com/JackRabbit911/Docker_php8.2 {{ ''|escape }}
    
    cd {{ ''|escape }}
    
    ./skeleton_install.sh

{{ __('Manually') }}:

1. {{ __('create_root_folder') }}:

    mkdir {{ ''|escape }}
    
    cd {{ ''|escape }}

2. {{ __('Use `curl` for download zip') }}:

    curl -L https://api.github.com/repos/JackRabbit911/app-skeleton/zipball/master > skeleton.zip
    
    unzip skeleton.zip
    
    mv JackRabbit911-app-skeleton* site.zone
    
    rm skeleton.zip

{{ __('Or use `git clone` for clone repository') }}:

    git clone https://github.com/JackRabbit911/app-skeleton site.zone
    
    rm -Rf site.zone/.git

3. {{ __('Go to site.zone folder') }}:

    cd site.zone

4. {{ __('Create writable directory') }} "storage":

    mkdir -p -m=777 storage
    
    mkdir -p -m=777 storage/logs storage/sessions storage/uploads storage/cache

5. {{ __('Create .env file') }}:

    cp .env.example .env

6. {{ __('Install framework') }}:

    composer require alpha-zeta/framework
